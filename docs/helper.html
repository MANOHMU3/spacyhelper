count_tags[source]

count_tags(spacy_data)

Count the occurence of each tag at token level. This is NOT SENT level

{% endraw %} {% raw %}
{% endraw %} 
spacy_data = [
    ("Who is Shaka Khan?", {"entities": [(7, 17, "PERSON")]}),
    ("I like London and Berlin.", {"entities": [(7, 13, "LOC"), (18, 24, "LOC")]}),
            ]
{% raw %} 
count_tags(spacy_data)

{'PERSON': 1, 'LOC': 2}
